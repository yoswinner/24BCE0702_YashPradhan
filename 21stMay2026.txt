# HAVING for WHERE for aggregate functions
select department, count(department) from worker
group by department having count(department)> 3 order by department asc;
select department, sum(salary)
from worker
group by department
having sum(salary) < 300000;

##
SELECT first_name, department, salary
FROM worker
WHERE salary = (
    SELECT DISTINCT salary
    FROM worker AS sub_worker
    WHERE sub_worker.department = worker.department
    ORDER BY salary ASC
    LIMIT 1 OFFSET 1
);

SELECT department, MIN(salary) AS second_lowest
FROM worker
WHERE salary > (
    SELECT MIN(salary) 
    FROM worker AS sub_worker 
    WHERE sub_worker.department = worker.department
)
GROUP BY department;

SELECT department, MIN(salary) AS lowest_salary
FROM worker
GROUP BY department;
