did int primary key,
dname varchar(25) not null unique
);

create table HOD
(Hid int primary key,
hname varchar(20) not null unique);

(105,'Murali');

SELECT * FROM 
DEPARTMENTS CROSS JOIN HOD ORDER BY DID;

SELECT * FROM 
DEPARTMENTS INNER JOIN HOD WHERE DEPARTMENTS.DID=HOD.HID;

SELECT * FROM 
DEPARTMENTS NATURAL JOIN HOD;

SELECT * FROM 
DEPARTMENTS LEFT OUTER JOIN HOD ON DEPARTMENTS.DID=HOD.HID ORDER BY DID ASC;

SELECT * FROM 
DEPARTMENTS RIGHT OUTER JOIN HOD ON DEPARTMENTS.DID=HOD.HID ORDER BY DID ASC;
